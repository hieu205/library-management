🧠 1. Vấn đề bạn đang giải quyết là gì?

👉 Cùng 1 trang /books/:id
nhưng:

Guest → xem sơ
User → xem chi tiết
Admin → có thêm nút quản lý
🧱 2. Cách làm chuẩn

👉 Tách thành 3 view riêng:

GuestView   → xem sơ
UserView    → xem full
AdminView   → full + sửa/xoá
🔗 3. Ai gọi mấy thằng này?

👉 Không phải router

👉 Mà là BookPage

🧩 4. Luồng hoạt động (rất quan trọng)
Route (/books/:id)
        ↓
BookDetail (page)
        ↓
BookPage (xử lý role)
        ↓
GuestView | UserView | AdminView
🧪 5. Code tối giản nhất
📄 BookPage.jsx
import GuestView from "./GuestView";
import UserView from "./UserView";
import AdminView from "./AdminView";

function BookPage({ role, book }) {
  if (role === "admin") return <AdminView book={book} />;
  if (role === "user") return <UserView book={book} />;
  return <GuestView book={book} />;
}

export default BookPage;
📄 BookDetail.jsx
import BookPage from "../components/book/BookPage";
import { useParams } from "react-router-dom";

function BookDetail() {
  const { id } = useParams();

  const user = JSON.parse(localStorage.getItem("user"));
  const role = user ? user.role : "guest";

  const book = {
    title: "Doraemon",
    shortDescription: "Truyện mèo máy",
    fullDescription: "Truyện dài...",
    author: "Fujiko"
  };

  return <BookPage role={role} book={book} />;
}
📄 GuestView.jsx
function GuestView({ book }) {
  return (
    <>
      <h2>{book.title}</h2>
      <p>{book.shortDescription}</p>
      <p>👉 Đăng nhập để xem chi tiết</p>
    </>
  );
}
📄 UserView.jsx
function UserView({ book }) {
  return (
    <>
      <h2>{book.title}</h2>
      <p>{book.fullDescription}</p>
      <p>{book.author}</p>
    </>
  );
}
📄 AdminView.jsx
function AdminView({ book }) {
  return (
    <>
      <h2>{book.title}</h2>
      <p>{book.fullDescription}</p>
      <button>Sửa</button>
      <button>Xoá</button>
    </>
  );
}
