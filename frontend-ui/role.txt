📄 BookPage.jsx
import GuestView from "./GuestView";
import UserView from "./UserView";
import AdminView from "./AdminView";

function BookPage({ role, book }) {
  if (role === "admin") return <AdminView book={book} />;
  if (role === "user") return <UserView book={book} />;
  return <GuestView book={book} />;
}

export default BookPage;

✅ 3. BookDetail.jsx (chỗ truyền data)
import BookPage from "../components/book/BookPage"
import { useParams } from "react-router-dom"
import { useEffect, useState } from "react"
import axios from "axios"

function BookDetail() {
  const { id } = useParams()

  const [book, setBook] = useState(null)

  // 👇 lấy role từ localStorage
  const user = JSON.parse(localStorage.getItem("user"))
  const role = user ? user.role : "guest"

  useEffect(() => {
    loadBook()
  }, [id])

  const loadBook = async () => {
    try {
      const res = await axios.get(
        `${import.meta.env.VITE_APP_URL}/books/${id}`
      )
      setBook(res.data.data)
    } catch (err) {
      console.error(err)
    }
  }

  if (!book) return <p>Loading...</p>

  return <BookPage role={role} book={book} />
}

export default BookDetail
